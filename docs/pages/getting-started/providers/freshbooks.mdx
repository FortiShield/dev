### Environment Variables

```
AUTH_FRESHBOOKS_ID
AUTH_FRESHBOOKS_SECRET
```

### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nextauth.js"
import FreshBooks from "nextauth.js/providers/freshbooks"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [FreshBooks],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@auth.ts"
import { QwikAuth$ } from "@nextauth.js/qwik"
import FreshBooks from "@nextauth.js/qwik/providers/freshbooks"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [FreshBooks],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nextauth.js/sveltekit"
import FreshBooks from "@nextauth.js/sveltekit/providers/freshbooks"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [FreshBooks],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nextauth.js/express"
import FreshBooks from "@nextauth.js/express/providers/freshbooks"

app.use("/auth/*", ExpressAuth({ providers: [FreshBooks] }))
```

  </Code.Express>
</Code>

### Notes

- Freshbooks requires an additional apiVersion parameter in YYYYMMDD format, which essentially states "I'm prepared for API changes up to this date".
