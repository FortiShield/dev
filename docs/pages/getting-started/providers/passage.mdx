### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nextauth.js"
import Passage from "nextauth.js/providers/passage"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Passage],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@auth.ts"
import { QwikAuth$ } from "@nextauth.js/qwik"
import Passage from "@nextauth.js/qwik/providers/passage"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [Passage],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nextauth.js/sveltekit"
import Passage from "@nextauth.js/sveltekit/providers/passage"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [Passage],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nextauth.js/express"
import Passage from "@nextauth.js/express/providers/passage"

app.use("/auth/*", ExpressAuth({ providers: [Passage] }))
```
