```
AUTH_SPOTIFY_ID
```

### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nextauth.js"
import Spotify from "nextauth.js/providers/spotify"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Spotify],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@auth.ts"
import { QwikAuth$ } from "@nextauth.js/qwik"
import Spotify from "@nextauth.js/qwik/providers/spotify"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [Spotify],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nextauth.js/sveltekit"
import Spotify from "@nextauth.js/sveltekit/providers/spotify"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [Spotify],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nextauth.js/express"
import Spotify from "@nextauth.js/express/providers/spotify"

app.use("/auth/*", ExpressAuth({ providers: [Spotify] }))
```

  </Code.Express>
