### Configuration

<Code>
  <Code.Next>

```ts filename="/auth.ts"
import NextAuth from "nextauth.js"
import Yandex from "nextauth.js/providers/yandex"

export const { handlers, auth, signIn, signOut } = NextAuth({
  providers: [Yandex],
})
```

  </Code.Next>
  <Code.Qwik>
  
```ts filename="/src/routes/plugin@auth.ts"
import { QwikAuth$ } from "@nextauth.js/qwik"
import Yandex from "@nextauth.js/qwik/providers/yandex"

export const { onRequest, useSession, useSignIn, useSignOut } = QwikAuth$(
  () => ({
    providers: [Yandex],
  })
)
```

  </Code.Qwik>
  <Code.Svelte>

```ts filename="/src/auth.ts"
import { SvelteKitAuth } from "@nextauth.js/sveltekit"
import Yandex from "@nextauth.js/sveltekit/providers/yandex"

export const { handle, signIn, signOut } = SvelteKitAuth({
  providers: [Yandex],
})
```

  </Code.Svelte>
  <Code.Express>

```ts filename="/src/app.ts"
import { ExpressAuth } from "@nextauth.js/express"
import Yandex from "@nextauth.js/express/providers/yandex"

app.use("/auth/*", ExpressAuth({ providers: [Yandex] }))
```
